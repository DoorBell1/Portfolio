% =========================================================================
% Digital AM Modulation and Demodulation System Pipeline
% Authored by: Simon Pham
% Core Architecture: Baseband Loading -> DSB-SC AM -> Multipath Nulling
% =========================================================================

clc; 
clear; 
close all; 

%% 1. Load Audio and Initialize DSP System Parameters
% Load high-resolution audio sample baseline pipeline
[y192, Fs192] = audioread("inputData/myName192kHz.wav");

freq_limit = 200;           % Optimization step frequency resolution limit
Fc = 20000;                 % 20 kHz Carrier frequency for AM translation
cutoff = 20000;             % 20 kHz low-pass bandwidth limit for vocal range
M = 501;                    % FIR filter length window coefficient parameters
delaySec = 0.1;             % Channel propagation delay simulation (seconds)
volume = 0.6;               % Echo path attenuation scale factor (α)

N = length(y192);
n = 0:N-1;
t192 = n / Fs192;           % Time vector definition

%% 2. DSB-SC Amplitude Modulation Stage
carrier = cos(2*pi*Fc*t192)';
modulated_signal = y192 .* carrier;

%% 3. Channel Impairments (Multipath Reflection / Delay Modeling)
delaySamples = round(delaySec * Fs192);
echo_signal = zeros(size(y192));
echo_signal(delaySamples+1:end) = y192(1:end-delaySamples);
corrupted_received = modulated_signal + (volume * echo_signal .* carrier);

%% 4. Coherent Demodulation & Inverse Rational Equalization Stage
demod_raw = corrupted_received .* carrier;

% Equalizer Matrix Definition: H(z) = 1 / (1 + alpha * z^-D)
% Formulate coefficients for concurrent Inverse IIR execution loop
b_eq = 1;
a_eq = zeros(1, delaySamples + 1);
a_eq(1) = 1;
a_eq(delaySamples + 1) = volume;

% Filter system parameters to reconstruct pristine channel output
equalized_demod = filter(b_eq, a_eq, demod_raw);

%% 5. Analytical Evaluation of System Transfer Function Matrices
% Generate composite discrete analysis arrays
N_fft = length(equalized_demod);
freqs = (-N_fft/2:N_fft/2-1) * (Fs192 / N_fft);
signal_spectrum = fftshift(fft(equalized_demod));
